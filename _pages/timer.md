// Î™©Ìëú Í¥ÄÎ†® Ìï®ÏàòÎì§ (timer.jsÏóê ÏóÜÎäî Í∏∞Îä•Îì§)
function addGoal() {
  const goalList = document.getElementById('goalList');
  const goalRow = document.createElement('div');
  goalRow.className = 'goal-row';
  goalRow.innerHTML = `
    <input type="checkbox" onchange="toggleGoal(this)">
    <input type="text" placeholder="Enter your goal">
    <input type="number" placeholder="25" min="1" max="999">
    <button class="remove-goal" onclick="removeGoal(this)">√ó</button>
  `;
  goalList.appendChild(goalRow);
  
  // ÏÉàÎ°ú Ï∂îÍ∞ÄÎêú ÌÖçÏä§Ìä∏ ÏûÖÎ†•Ïóê Ïù¥Î≤§Ìä∏ Î¶¨Ïä§ÎÑà Ï∂îÍ∞Ä
  const textInput = goalRow.querySelector('input[type="text"]');
  handleGoalInput(textInput);
  textInput.focus(); // ÏûêÎèô Ìè¨Ïª§Ïä§
  
  // localStorage Ï†ÄÏû• (timer.jsÏùò saveToLocal Ìï®Ïàò ÌôúÏö©)
  if (typeof saveToLocal === 'function') {
    saveToLocal();
  }
}

function toggleGoal(checkbox) {
  const row = checkbox.parentElement;
  if (checkbox.checked) {
    row.classList.add('checked');
  } else {
    row.classList.remove('checked');
  }
  
  // localStorage Ï†ÄÏû•
  if (typeof saveToLocal === 'function') {
    saveToLocal();
  }
}

function removeGoal(button) {
  button.parentElement.remove();
  
  // localStorage Ï†ÄÏû•
  if (typeof saveToLocal === 'function') {
    saveToLocal();
  }
}

// Î™©Ìëú ÏûÖÎ†• ÏôÑÎ£å Ïãú ÏùΩÍ∏∞ Î™®ÎìúÎ°ú Ï†ÑÌôò
function handleGoalInput(input) {
  // ÏóîÌÑ∞ÌÇ§ Ï≤òÎ¶¨
  input.addEventListener('keypress', function(e) {
    if (e.key === 'Enter' && input.value.trim()) {
      finalizeGoal(input);
    }
  });
  
  // Ìè¨Ïª§Ïä§ ÏïÑÏõÉ Ï≤òÎ¶¨
  input.addEventListener('blur', function() {
    if (input.value.trim()) {
      finalizeGoal(input);
    }
  });
}

function finalizeGoal(input) {
  const goalRow = input.parentElement;
  goalRow.classList.add('readonly');
  input.blur();
  
  // localStorage Ï†ÄÏû•
  if (typeof saveToLocal === 'function') {
    saveToLocal();
  }
  
  // ÌÅ¥Î¶≠ Ïãú Îã§Ïãú Ìé∏Ïßë Í∞ÄÎä•ÌïòÍ≤å
  input.addEventListener('click', function editGoal() {
    goalRow.classList.remove('readonly');
    input.focus();
    input.removeEventListener('click', editGoal); // Ïù¥Î≤§Ìä∏ Î¶¨Ïä§ÎÑà Ï†úÍ±∞
  });
}

// DOMÏù¥ Î°úÎìúÎêú ÌõÑ Ïã§Ìñâ
document.addEventListener('DOMContentLoaded', function() {
  // Í∏∞Î≥∏ÏúºÎ°ú Î™©Ìëú 3Í∞ú ÏÉùÏÑ±
  for (let i = 0; i < 3; i++) {
    addGoal();
  }
  
  // ÌéòÏù¥ÏßÄ Î°úÎìú Ïãú Í∏∞Ï°¥ Î™©ÌëúÎì§Ïóê Ïù¥Î≤§Ìä∏ Î¶¨Ïä§ÎÑà Ï∂îÍ∞Ä
  setTimeout(function() {
    const goalInputs = document.querySelectorAll('.goal-row input[type="text"]:not([data-enhanced])');
    goalInputs.forEach(input => {
      input.setAttribute('data-enhanced', 'true');
      handleGoalInput(input);
    });
  }, 100); // timer.jsÏùò loadFromLocal Ïã§Ìñâ ÌõÑ
});
